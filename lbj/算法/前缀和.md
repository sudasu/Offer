# 前缀和

```golang
func subarraySum(nums []int, k int) int {
    store := make(map[int]int)
    rst := 0
    count := 0
    store[0] = 1 // 保证数组只有一个数时能统计进去
    for i := 0 ; i < len(nums) ; i ++ {
        count += nums[i]
        rst += store[count - k]
        store[count] += 1
    }
    return rst
}
```
